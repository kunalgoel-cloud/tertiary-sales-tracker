        function handleFileSelect(e) {

            const file = e.target.files[0];

            if (!file) return;


            const label = document.getElementById('fileLabel');

            const preview = document.getElementById('preview');

            const progressBar = document.getElementById('progressBar');

            const progressFill = document.getElementById('progressFill');

            const ocrStatus = document.getElementById('ocrStatus');

            const processing = document.getElementById('processing');

            const submitBtn = document.getElementById('submitBtn');


            // Show preview

            const reader = new FileReader();

            reader.onload = (e) => {

                preview.src = e.target.result;

                preview.style.display = 'block';

                label.classList.add('has-file');

                label.innerHTML = '<div style="color: var(--success); font-weight: 600;">✓ Image loaded</div>';

            };

            reader.readAsDataURL(file);


            // Process OCR

            processing.style.display = 'block';

            progressBar.style.display = 'block';

            ocrStatus.textContent = 'Initializing OCR engine...';

            submitBtn.disabled = true;


            Tesseract.recognize(

                file,

                'eng',

                {

                    logger: m => {

                        if (m.status === 'recognizing text') {

                            progressFill.style.width = (m.progress * 100) + '%';

                            ocrStatus.textContent = `Processing: ${Math.round(m.progress * 100)}%`;

                        }

                    }

                }

            ).then(({ data: { text } }) => {

                processing.style.display = 'none';

                progressBar.style.display = 'none';

                ocrStatus.textContent = 'OCR Complete! Review data before submitting.';

                submitBtn.disabled = false;

                

                extractedData = parseOCRText(text);

                

                if (extractedData.length > 0) {

                    showToast(`Found ${extractedData.length} items in image`, 'success');

                    const manualText = extractedData.map(d => `${d.item}, ${d.qty}, ${d.revenue}`).join('\n');

                    document.getElementById('manualData').value = manualText;

                    showToast('No data extracted. Please enter manually.', 'error');

                }

                submitBtn.disabled = false;

                document.getElementById('manualEntry').style.display = 'block';

                showToast('OCR processing failed', 'error');

            });

        }


        function parseOCRText(text) {

            const lines = text.split('\n').filter(line => line.trim());

            const results = [];

            

            const patterns = [

                /([A-Za-z\s]+)\s+(\d+)\s+(\d+(?:\.\d{2})?)/,

                /([A-Za-z\s]+)[,|\s]+(\d+)[,|\s]+(\d+(?:\.\d{2})?)/,

                /(\d+)\s+([A-Za-z\s]+)\s+(\d+(?:\.\d{2})?)/,

            ];


            lines.forEach(line => {

                for (let pattern of patterns) {

                    const match = line.match(pattern);

                    if (match) {

                        const item = match[1].trim();

                        const qty = parseInt(match[2]) || 0;

                        const revenue = parseFloat(match[3]) || 0;

                        if (item && qty > 0) {

                            results.push({ item, qty, revenue });

                            break;

                        }

                    }

                }

            });


            return results;

        }

        function handleSubmit(e) {

            e.preventDefault();

            

            if (!db) {

                showToast('Not connected to database. Please configure Firebase first.', 'error');

                return;

            }


            const channel = document.getElementById('channel').value;

            const date = document.getElementById('date').value;

            const manualData = document.getElementById('manualData').value;


            if (!channel || !date) {

                showToast('Please fill all required fields', 'error');

                return;

            }


            let items = [];


            if (extractedData.length > 0) {

                items = extractedData;

            } else if (manualData.trim()) {

                const lines = manualData.split('\n');

                lines.forEach(line => {

                    const parts = line.split(',').map(p => p.trim());

                    if (parts.length >= 3) {

                        items.push({

                            item: parts[0],

                            qty: parseInt(parts[1]) || 0,

                            revenue: parseFloat(parts[2]) || 0

                        });

                    }

                });

            }


            if (items.length === 0) {

                showToast('No valid data found to save', 'error');

                return;

            }


            // Save to Firebase

            const updates = {};

            items.forEach(item => {

                const newKey = db.ref().child('sales').push().key;

                updates['/sales/' + newKey] = {

                    id: newKey,

                    date: date,

                    channel: channel,

                    itemName: item.item,

                    qtySold: item.qty,

                    revenue: item.revenue,

                    timestamp: firebase.database.ServerValue.TIMESTAMP,

                    submittedBy: navigator.userAgent.split(' ')[0] // Simple device identifier

                };

            });


            db.ref().update(updates)

                .then(() => {

                    // Reset form

                    document.getElementById('salesForm').reset();

                    document.getElementById('preview').style.display = 'none';

                    document.getElementById('fileLabel').classList.remove('has-file');

                    document.getElementById('fileLabel').innerHTML = `

                        <div style="font-size: 2rem; margin-bottom: 10px;">📷</div>

                        <div>Click to take photo or upload image</div>

                        <div style="font-size: 0.9rem; opacity: 0.7; margin-top: 5px;">Supports: JPG, PNG, WEBP</div>

                    `;

                    document.getElementById('manualEntry').style.display = 'none';

                    document.getElementById('submitBtn').disabled = true;

                    extractedData = [];


                    showToast(`Saved ${items.length} items to cloud!`, 'success');

                })

                .catch(error => {

                    showToast('Error saving data: ' + error.message, 'error');

                });

        }


        function deleteEntry(id) {

            if (!confirm('Are you sure you want to delete this entry?')) return;

            

            db.ref('sales/' + id).remove()

                .then(() => showToast('Entry deleted', 'success'))

                .catch(err => showToast('Error deleting: ' + err.message, 'error'));

        }


        function clearAllData() {

            if (!confirm('⚠️ WARNING: This will delete ALL data for everyone. Are you sure?')) return;

            if (!confirm('Really sure? This action cannot be undone!')) return;

            

            db.ref('sales').remove()

                .then(() => showToast('All data cleared', 'success'))

                .catch(err => showToast('Error clearing data: ' + err.message, 'error'));

        }


        function updateItemFilter() {

            const items = [...new Set(salesData.map(d => d.itemName))].sort();

            const select = document.getElementById('filterItem');

            const currentVal = select.value;

            select.innerHTML = '<option value="all">All Items</option>';

            items.forEach(item => {

                select.innerHTML += `<option value="${item}">${item}</option>`;

            });

            if (items.includes(currentVal)) select.value = currentVal;

        }


        function getFilteredData() {

            const channel = document.getElementById('filterChannel').value;

            const item = document.getElementById('filterItem').value;

            const fromDate = document.getElementById('filterDateFrom').value;

            const toDate = document.getElementById('filterDateTo').value;


            return salesData.filter(d => {

                if (channel !== 'all' && d.channel !== channel) return false;

                if (item !== 'all' && d.itemName !== item) return false;

                if (fromDate && d.date < fromDate) return false;

                if (toDate && d.date > toDate) return false;

                return true;

            });

        }


        function updateDashboard() {

            const filtered = getFilteredData();

            

            const totalRevenue = filtered.reduce((sum, d) => sum + (d.revenue || 0), 0);

            const totalQty = filtered.reduce((sum, d) => sum + (d.qtySold || 0), 0);

            const totalTransactions = filtered.length;

            const avgOrder = totalTransactions > 0 ? totalRevenue / totalTransactions : 0;


            document.getElementById('totalRevenue').textContent = '₹' + Math.round(totalRevenue).toLocaleString();

            document.getElementById('totalQty').textContent = totalQty.toLocaleString();

            document.getElementById('totalTransactions').textContent = totalTransactions.toLocaleString();

            document.getElementById('avgOrderValue').textContent = '₹' + Math.round(avgOrder).toLocaleString();


            updateChart(filtered);

            updateTable(filtered);

        }

        function switchTab(tab) {

            document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));

            document.querySelectorAll('.tab-content').forEach(t => t.classList.remove('active'));

            

            event.target.classList.add('active');

            }

        }


        function downloadCSV() {

            const filtered = getFilteredData();

            if (filtered.length === 0) {

                showToast('No data to download', 'error');

                return;

            }


            const headers = ['Date', 'Channel', 'Item Name', 'Qty Sold', 'Revenue'];

            const csvContent = [

                headers.join(','),

                ...filtered.map(d => [

                    d.date,

                    d.channel,

                    `"${d.itemName}"`,

                    d.qtySold,

                    Math.round(d.revenue)

                ].join(','))

            ].join('\n');


            const blob = new Blob([csvContent], { type: 'text/csv' });

            const url = window.URL.createObjectURL(blob);

            const a = document.createElement('a');

            a.href = url;

            a.download = `sales_data_${new Date().toISOString().split('T')[0]}.csv`;

            document.body.appendChild(a);

            a.click();

            document.body.removeChild(a);

            window.URL.revokeObjectURL(url);


            showToast('CSV downloaded successfully!', 'success');

        }


        function showToast(message, type = 'success') {

            const toast = document.getElementById('toast');

            const icon = document.getElementById('toastIcon');

            const msg = document.getElementById('toastMessage');

            

            icon.textContent = type === 'success' ? '✅' : '❌';

            msg.textContent = message;

            toast.className = 'toast' + (type === 'error' ? ' error' : '');

            toast.style.display = 'flex';

            

            setTimeout(() => {

                toast.style.display = 'none';

            }, 3000);

        }
